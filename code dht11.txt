#include <dht.h>

dht DHT ;                       // declaring dht a variable

#define DHT11_PIN D4             // initializing pin 8 for dht

void setup ( ) {                                                

  Serial.begin ( 9600 ) ;       // starting the 16 x 2 lcd

}

void loop ( )

{

  int chk = DHT.read11(DHT11_PIN ) ;         // Checking that either the dht is working or not

               //  starting the cursor from  top left

  Serial.print ( " Temperature is : " ) ;       // printing the “ Temperature is : ” on the lcd 

  Serial.print ( DHT.temperature ) ;            // printing the temperature on the lcd
                     

  Serial.print ( " Humidity is : " ) ;         // printing “ humidity is : ” on the display

  Serial.print ( DHT.humidity ) ;              // printing humidity on the display

 Serial.print ( " % " ) ;                     // printing “ % ” on display

  delay ( 1000 ) ;                          // Giving delay of 1 second.

}
